// =======================================================
// [설정] 시트 ID 및 시트 이름
// =======================================================
const SPREADSHEET_ID = "시트주소"; 

const SHEET_NAME_LOG = "정품등록_고객정보"; 
const SHEET_NAME_CLICK_LOG = "프로모션_클릭로그"; 
const SHEET_NAME_STORE = "거래처정보";   
const SHEET_NAME_PRODUCT = "제품정보";
const SHEET_NAME_SERIAL = "차대번호정보"; 

// [보안] 클라이언트(js/common.js)와 일치해야 하는 비밀 토큰
const API_TOKEN = "Token"; 

// =======================================================
// [1] GET/POST 요청 분기
// =======================================================
function doGet(e) {
  if (!e.parameter.token || e.parameter.token !== API_TOKEN) {
    return responseJSON({ status: "error", message: "Unauthorized: 접근 권한이 없습니다." });
  }

  const params = e.parameter;
  if (params.type === "check") return checkSerial(params.no);
  if (params.type === "search") return searchRegistration(params.name, params.phone);
  return getInitData();
}

function doPost(e) {
  const lock = LockService.getScriptLock();
  try {
    lock.waitLock(30000); 
    
    if (!e.parameter.token || e.parameter.token !== API_TOKEN) {
       return responseJSON({ result: "error", message: "Unauthorized: 접근 권한이 없습니다." });
    }

    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.openById(SPREADSHEET_ID);

    // [기능 A] 프로모션 클릭 로그
    if (data.type === "log_click") {
      const clickSheet = ss.getSheetByName(SHEET_NAME_CLICK_LOG);
      if (clickSheet) {
        const now = new Date();
        const clickId = "CLK-" + now.getTime().toString(36).toUpperCase() + Math.floor(Math.random() * 1000);
        clickSheet.appendRow([
          clickId, now, data.name, data.phone, data.regId, data.storeName, data.storeCode, data.model
        ]);
      }
      return responseJSON({ result: "success" });
    }

    // [기능 B] 제품 등록 실행
    const logSheet = ss.getSheetByName(SHEET_NAME_LOG);
    const serialSheet = ss.getSheetByName(SHEET_NAME_SERIAL);
    
    const checkResult = checkSerial(data.serialNo);
    const checkJson = JSON.parse(checkResult.getContent());
    if (checkJson.status !== "ok") {
       return responseJSON({ result: "error", message: checkJson.message });
    }

    const now = new Date();
    const regId = generateRegId(now); 

    // ── 구글 드라이브 부모 폴더 탐색 (스프레드시트와 같은 위치) ──
    const ssFile = DriveApp.getFileById(SPREADSHEET_ID);
    const parents = ssFile.getParents();
    const parentFolder = parents.hasNext() ? parents.next() : DriveApp.getRootFolder();

    // ── [NEW] 영수증 이미지 업로드 ──
    let fileUrl = "";
    if (data.fileData) {
      const receiptFolderName = "제품등록_영수증";
      const receiptFolders = parentFolder.getFoldersByName(receiptFolderName);
      let receiptFolder = receiptFolders.hasNext() ? receiptFolders.next() : parentFolder.createFolder(receiptFolderName);
      
      const originalName = data.fileName || "receipt";
      const ext = originalName.includes('.') ? originalName.split('.').pop() : 'jpg'; 
      const receiptFileName = `${regId}_${data.userName}_${data.product}_${data.storeName}_${data.serialNo}.${ext}`;

      const receiptBlob = Utilities.newBlob(Utilities.base64Decode(data.fileData), data.mimeType, receiptFileName);
      const receiptFile = receiptFolder.createFile(receiptBlob);
      receiptFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      fileUrl = receiptFile.getUrl();
    }

    // ── [NEW] 차대번호 사진 업로드 ──
    let serialImageUrl = "";
    if (data.serialImageData) {
      const serialFolderName = "제품등록_차대번호사진";
      const serialFolders = parentFolder.getFoldersByName(serialFolderName);
      let serialFolder = serialFolders.hasNext() ? serialFolders.next() : parentFolder.createFolder(serialFolderName);
      
      const serialFileName = `${regId}_차대번호_${data.serialNo}.jpg`;
      const serialBlob = Utilities.newBlob(Utilities.base64Decode(data.serialImageData), "image/jpeg", serialFileName);
      const serialFile = serialFolder.createFile(serialBlob);
      serialFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      serialImageUrl = serialFile.getUrl();
    }

    const personal = "동의함"; 
    const thirdParty = "동의함"; 
    const transfer = data.transferConsent ? "동의함" : "동의안함"; 
    const marketing = data.marketingConsent ? "동의함" : "동의안함";
    const specialApply = ""; // 관리자 특별적용 (등록 시점엔 항상 빈값)

    // ── 스프레드시트 열 순서: A~O ──
    // A:등록ID B:이름 C:연락처 D:모델 E:매장 F:거래처코드 G:차대번호
    // H:영수증이미지링크 I:개인정보수집 J:제3자제공 K:국외이전 L:마케팅
    // M:등록일시 N:관리자특별적용 O:차대번호이미지링크
    logSheet.appendRow([
      regId, data.userName, data.userPhone, data.product, 
      data.storeName, data.storeCode, data.serialNo, 
      fileUrl,                                          // H: 영수증 이미지 링크
      personal, thirdParty, transfer, marketing, now,  // I~M
      specialApply,                                    // N: 관리자 특별적용
      serialImageUrl                                   // O: 차대번호 사진 링크 [NEW]
    ]);

    // ── 차대번호 시트 '사용완료' 처리 ──
    if (serialSheet) {
      const lastRow = serialSheet.getLastRow();
      const serialData = serialSheet.getRange(2, 1, lastRow - 1, 1).getValues().flat();
      const rowIndex = serialData.findIndex(s => s.toString().trim().toUpperCase() === data.serialNo.toString().trim().toUpperCase());
      if (rowIndex !== -1) {
        serialSheet.getRange(rowIndex + 2, 4).setValue("사용완료");
      }
    }
    return responseJSON({ result: "success" });

  } catch (error) {
    return responseJSON({ result: "error", message: error.toString() });
  } finally {
    lock.releaseLock();
  }
}

// [기능 1] 내역 조회 (특별 적용 로직 포함)
// 스프레드시트 열 순서 변경됨: N열(인덱스13)=관리자특별적용, O열(인덱스14)=차대번호이미지링크
function searchRegistration(name, phone) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const sheet = ss.getSheetByName(SHEET_NAME_LOG);
  const serialSheet = ss.getSheetByName(SHEET_NAME_SERIAL);
  const storeSheet = ss.getSheetByName(SHEET_NAME_STORE);

  if (!sheet) return responseJSON({ status: "fail", message: "DB 없음" });
  
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return responseJSON({ status: "fail", message: "데이터 없음" });

  // N열(특별적용)까지 읽음 — 인덱스 0~13, 총 14열
  const data = sheet.getRange(2, 1, lastRow - 1, 14).getValues();
  
  let serialYearMap = {};
  if (serialSheet && serialSheet.getLastRow() >= 2) {
    const sData = serialSheet.getRange(2, 1, serialSheet.getLastRow() - 1, 2).getValues();
    sData.forEach(r => { if(r[0]) serialYearMap[r[0].toString().trim().toUpperCase()] = r[1]; });
  }

  let storeInfoMap = {};
  if (storeSheet && storeSheet.getLastRow() >= 2) {
    const stData = storeSheet.getRange(2, 1, storeSheet.getLastRow() - 1, 8).getValues();
    stData.forEach(r => {
      if(r[1]) {
        storeInfoMap[r[1].toString().trim()] = {
          isCredit: (r[6] === "여신거래대상"),
          promoUrl: (r[7] && r[7].toString().trim() !== "") ? r[7].toString().trim() : null
        };
      }
    });
  }

  const targetName = name.toString().trim();
  const targetPhone = phone.toString().trim().replace(/-/g, ""); 
  let results = [];

  for (let i = data.length - 1; i >= 0; i--) {
    let row = data[i];
    let dbName = row[1].toString().trim();
    let dbPhone = row[2].toString().trim().replace(/-/g, "");

    if (dbName === targetName && dbPhone === targetPhone) {
      const serialKey = row[6].toString().trim().toUpperCase();
      let year = serialYearMap[serialKey] ? serialYearMap[serialKey].toString() : "2026";
      
      const storeName = row[4].toString().trim();
      const storeInfo = storeInfoMap[storeName] || { isCredit: false, promoUrl: null };
      
      const marketingStatus = row[11] ? row[11].toString() : "동의안함";
      const regDate = row[12];
      
      // N열(인덱스 13): 관리자 특별적용
      const adminSpecial = row[13] ? row[13].toString().trim() : "";
      let isSpecial = false;
      
      if (adminSpecial === "적용") {
        year = "2026";
        isSpecial = true;
      }

      results.push({
        id: row[0], name: row[1], phone: row[2], product: row[3], store: row[4], storeCode: row[5], serial: row[6],    
        date: regDate, year: year, isCredit: storeInfo.isCredit, storeLink: storeInfo.promoUrl,
        marketing: marketingStatus,
        isSpecial: isSpecial
      });
    }
  }

  if (results.length > 0) return responseJSON({ status: "success", data: results });
  else return responseJSON({ status: "fail", message: "일치하는 정보 없음" });
}

function checkSerial(serialNo) {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const target = serialNo.toString().trim().toUpperCase();
  if (target.length < 3) return responseJSON({ status: "fail", message: "입력 오류" });

  const masterSheet = ss.getSheetByName(SHEET_NAME_SERIAL);
  if (masterSheet && masterSheet.getLastRow() >= 2) {
    const masterData = masterSheet.getRange(2, 1, masterSheet.getLastRow() - 1, 4).getValues();
    const foundRow = masterData.find(row => row[0].toString().trim().toUpperCase() === target);
    
    if (!foundRow) return responseJSON({ status: "fail", message: "유효하지 않은 차대번호입니다." });
    if (foundRow[3] && foundRow[3].toString() === "사용완료") return responseJSON({ status: "fail", message: "이미 등록 완료된 차대번호입니다." });
    
    return responseJSON({ status: "ok", message: "제품 확인 완료", model: foundRow[2] });
  } else {
    return responseJSON({ status: "fail", message: "제품 DB 비어있음" });
  }
}

function getInitData() {
  const ss = SpreadsheetApp.openById(SPREADSHEET_ID);
  const storeSheet = ss.getSheetByName(SHEET_NAME_STORE);
  let storeList = [];
  if (storeSheet && storeSheet.getLastRow() > 1) {
    const data = storeSheet.getRange(2, 1, storeSheet.getLastRow() - 1, 6).getValues();
    storeList = data.map(row => ({
      code: row[0].toString(), name: row[1].toString(), phone: row[2] ? row[2].toString() : "", 
      addr: row[3] ? row[3].toString() : "", alias: row[4] ? row[4].toString() : "", agency: row[5] ? row[5].toString() : "", 
    })).filter(s => s.name);
  }
  const productSheet = ss.getSheetByName(SHEET_NAME_PRODUCT);
  let productList = [];
  if (productSheet && productSheet.getLastRow() > 1) {
    const data = productSheet.getRange(2, 1, productSheet.getLastRow() - 1, 1).getValues();
    productList = data.flat().filter(String);
  }
  return responseJSON({ products: productList, stores: storeList });
}

function generateRegId(date) {
  const yyyy = date.getFullYear();
  const mm = String(date.getMonth() + 1).padStart(2, '0');
  const dd = String(date.getDate()).padStart(2, '0');
  const randomNum = Math.floor(Math.random() * 10000).toString().padStart(4, '0');
  return `XTRON_REG-${yyyy}${mm}${dd}_${randomNum}`;
}

function responseJSON(data) {
  return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON);
}
